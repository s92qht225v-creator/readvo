```typescript
import jwt from 'jsonwebtoken';
import { config } from '../config.js';

interface JwtPayload {
  sub: string;
  email: string;
  iat: number;
  exp: number;
}

export function signToken(userId: string, email: string, expiresIn: string = '7d'): string {
  return jwt.sign({ sub: userId, email }, config.jwtSecret, { expiresIn });
}

export function verifyToken(token: string): JwtPayload {
  return jwt.verify(token, config.jwtSecret) as JwtPayload;
}

export function refreshToken(token: string): string {
  const payload = verifyToken(token);
  return signToken(payload.sub, payload.email);
}

export function decodeToken(token: string): JwtPayload | null {
  try {
    return jwt.decode(token) as JwtPayload | null;
  } catch {
    return null;
  }
}
```

```typescript
import type { Response, NextFunction } from 'express';
import { verifyToken, decodeToken } from '../lib/jwt.js';
import type { AuthRequest } from '../types.js';

/** Extracts userId from JWT if present. Does NOT reject unauthenticated requests. */
export function extractAuth(req: AuthRequest, _res: Response, next: NextFunction): void {
  const header = req.headers.authorization;
  const token = header?.startsWith('Bearer ') ? header.slice(7) : null;

  if (token) {
    try {
      const payload = verifyToken(token);
      req.userId = payload.sub;
      req.token = token;
    } catch {
      // Token invalid/expired — treat as unauthenticated
    }
  }
  next();
}

/** Rejects unauthenticated requests with 401. Use after extractAuth. */
export function requireAuth(req: AuthRequest, res: Response, next: NextFunction): void {
  if (!req.userId) {
    res.status(401).json({ error: 'Authentication required' });
    return;
  }
  next();
}
```

```typescript
import OpenAI from 'openai';
import { config } from '../config.js';
import type { ScoreResult } from '../types.js';

const openai = new OpenAI({ apiKey: config.openaiApiKey });

const CRITICAL_CHARS = new Set('我你他她它这那有没不是很都也吗呢吧啊'.split(''));

export function normalize(str: string): string {
  let s = str.trim().toLowerCase();
  s = s.replace(/[。？！，、：；""''「」《》（）()\s\d]/g, '');
  s = s.replace(/她/g, '他').replace(/它/g, '他');
  return s;
}

export function levenshtein(a: string, b: string): number {
  const m = a.length, n = b.length;
  const dp: number[][] = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0));
  for (let i = 0; i <= m; i++) dp[i][0] = i;
  for (let j = 0; j <= n; j++) dp[0][j] = j;
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      dp[i][j] = a[i - 1] === b[j - 1]
        ? dp[i - 1][j - 1]
        : 1 + Math.min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]);
    }
  }
  return dp[m][n];
}

function hasCriticalCharChange(expected: string, heard: string): boolean {
  const eSet = new Set(expected.split(''));
  const hSet = new Set(heard.split(''));
  for (const c of CRITICAL_CHARS) {
    if (eSet.has(c) !== hSet.has(c)) return true;
  }
  return false;
}

async function askJudge(expected: string, heard: string, language: string): Promise<ScoreResult> {
  const langLabel = language === 'uz' ? 'Uzbek (Latin script only, NOT Cyrillic)'
    : language === 'ru' ? 'Russian' : 'English';

  try {
    const response = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      temperature: 0,
      max_tokens: 80,
      messages: [
        {
          role: 'system',
          content: `You grade Chinese speech practice. Compare Expected vs Heard.
Return JSON: {"result":"correct"|"close"|"wrong","feedback":"..."}
CORRECT: exact or learner said expected + extra context words
CLOSE: tone-only mistake (e.g. 那→拿) or minor noise, meaning clear
WRONG: substitution changing meaning (e.g. 我→你, 不→没)
Name the specific wrong/missing character.
IMPORTANT: feedback MUST be in ${langLabel} language only.`,
        },
        { role: 'user', content: `Expected: ${expected}\nHeard: ${heard}` },
      ],
    });

    const text = response.choices[0]?.message?.content?.trim() || '';
    const json = JSON.parse(text);
    return { result: json.result, feedback: json.feedback };
  } catch {
    const dist = levenshtein(normalize(expected), normalize(heard));
    return dist <= 1
      ? { result: 'close', feedback: '' }
      : { result: 'wrong', feedback: '' };
  }
}

export async function scoreAnswer(
  expected: string,
  heard: string,
  language: string
): Promise<ScoreResult> {
  const ne = normalize(expected);
  const nh = normalize(heard);

  if (ne === nh) return { result: 'correct', feedback: '' };
  if (nh.includes(ne)) return { result: 'correct', feedback: '' };

  const dist = levenshtein(ne, nh);

  if (hasCriticalCharChange(ne, nh)) {
    return askJudge(expected, heard, language);
  }

  if (ne.length <= 4) {
    return askJudge(expected, heard, language);
  }

  if (dist <= 1) return { result: 'correct', feedback: '' };
  if (dist <= 4) return askJudge(expected, heard, language);
  return { result: 'wrong', feedback: '' };
}
```

```typescript
import { config } from '../config.js';
import type { WhisperResult } from '../types.js';

function buildPrompt(expected?: string): string {
  let prompt = '用简体中文输出。';
  if (expected) {
    const chars = [...new Set(expected.replace(/[^\u4e00-\u9fff]/g, '').split(''))];
    if (chars.length > 0) {
      prompt += `上下文词汇：${chars.join('、')}`;
    }
  }
  return prompt;
}

async function tryGpt4o(
  audioBuffer: ArrayBuffer,
  fileName: string,
  mimeType: string,
  prompt: string
): Promise<WhisperResult | null> {
  try {
    const form = new FormData();
    form.append('file', new Blob([audioBuffer], { type: mimeType }), fileName);
    form.append('model', 'gpt-4o-transcribe');
    form.append('language', 'zh');
    form.append('prompt', prompt);
    form.append('response_format', 'json');

    const res = await fetch('https://api.openai.com/v1/audio/transcriptions', {
      method: 'POST',
      headers: { Authorization: `Bearer ${config.openaiApiKey}` },
      body: form,
    });

    if (!res.ok) return null;
    const data = await res.json();
    return data.text ? { text: data.text, source: 'gpt4o', noSpeechProb: 0 } : null;
  } catch {
    return null;
  }
}

async function tryGroq(
  audioBuffer: ArrayBuffer,
  fileName: string,
  mimeType: string,
  prompt: string
): Promise<WhisperResult | null> {
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 3000);

    const form = new FormData();
    form.append('file', new Blob([audioBuffer], { type: mimeType }), fileName);
    form.append('model', 'whisper-large-v3');
    form.append('language', 'zh');
    form.append('prompt', prompt);
    form.append('response_format', 'verbose_json');

    const res = await fetch('https://api.groq.com/openai/v1/audio/transcriptions', {
      method: 'POST',
      headers: { Authorization: `Bearer ${config.groqApiKey}` },
      body: form,
      signal: controller.signal,
    });

    clearTimeout(timeout);
    if (!res.ok) return null;

    const data = await res.json();
    const segments = data.segments || [];
    const avgNoSpeech = segments.length > 0
      ? segments.reduce((sum: number, s: any) => sum + (s.no_speech_prob || 0), 0) / segments.length
      : 0;

    return data.text ? { text: data.text, source: 'groq', noSpeechProb: avgNoSpeech } : null;
  } catch {
    return null;
  }
}

export async function transcribeAudio(
  audioBuffer: ArrayBuffer,
  fileName: string,
  mimeType: string,
  expected?: string
): Promise<WhisperResult | null> {
  const prompt = buildPrompt(expected);
  const result = await tryGpt4o(audioBuffer, fileName, mimeType, prompt);
  if (result) return result;
  return tryGroq(audioBuffer, fileName, mimeType, prompt);
}
```

```typescript
import OpenAI from 'openai';
import { config } from '../config.js';
import { levenshtein, normalize } from './scorer.js';

const openai = new OpenAI({ apiKey: config.openaiApiKey });

function sharedCharRatio(a: string, b: string): number {
  const setA = new Set(a.split(''));
  let shared = 0;
  for (const c of b.split('')) {
    if (setA.has(c)) shared++;
  }
  return shared / Math.max(a.length, b.length);
}

function isChinese(str: string): boolean {
  return /[\u4e00-\u9fff]/.test(str);
}

export async function postCorrect(expected: string, heard: string): Promise<string> {
  const ne = normalize(expected);
  const nh = normalize(heard);
  const dist = levenshtein(ne, nh);

  // Only correct if distance 2+ but not too far, and enough shared chars
  if (dist < 2 || dist > ne.length || sharedCharRatio(ne, nh) < 0.6) {
    return heard;
  }

  try {
    const response = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      temperature: 0,
      max_tokens: 100,
      messages: [
        {
          role: 'system',
          content: `Fix Chinese homophone errors. The learner meant to say Expected but STT heard Heard.
If Heard has wrong characters that SOUND THE SAME as Expected characters, fix them.
If Heard has a DIFFERENT meaning or different sounds, return Heard UNCHANGED.
Return ONLY the corrected Chinese text, nothing else.`,
        },
        { role: 'user', content: `Expected: ${expected}\nHeard: ${heard}` },
      ],
    });

    const corrected = response.choices[0]?.message?.content?.trim() || '';
    if (!corrected || corrected.length > heard.length * 2 || !isChinese(corrected)) {
      return heard;
    }
    return corrected;
  } catch {
    return heard;
  }
}
```

```typescript
    const name = payload.name || email.split('@')[0];

        return;
      }
      const { data: otpData } = await admin.auth.verifyOtp({
        type: 'magiclink',
        token_hash: linkData.properties.hashed_token,
      });
      userId = otpData?.user?.id || '';
      // Update metadata
      await admin.auth.admin.updateUserById(userId, {
        user_metadata: { name, avatar_url: avatar, provider: 'google' },
      });
    } else if (createData?.user) {
      userId = createData.user.id;
    } else {
      res.status(500).json({ error: 'Failed to create user' });
      return;
    }

    // Session nonce
    const nonce = crypto.randomBytes(16).toString('hex');
    await admin.from('active_sessions').upsert(
      { user_id: userId, session_nonce: nonce, updated_at: new Date().toISOString() },
```

```typescript
import { Router } from 'express';
import multer from 'multer';
import { extractAuth, requireAuth } from '../middleware/auth.js';
import { getSupabase } from '../lib/supabase.js';
import { transcribeAudio } from '../transcribe/whisper.js';
import { scoreAnswer, normalize, levenshtein } from '../transcribe/scorer.js';
import { postCorrect } from '../transcribe/post-correct.js';
import type { AuthRequest, TranscribeResponse } from '../types.js';

const router = Router();
const upload = multer({ storage: multer.memoryStorage(), limits: { fileSize: 10 * 1024 * 1024 } });

function isChinese(str: string): boolean {
  return /[\u4e00-\u9fff]/.test(str);
}

function isLatin(str: string): boolean {
  return /^[a-zA-Z\s.,!?]+$/.test(str.trim());
}

router.post('/', upload.single('audio'), extractAuth, requireAuth, async (req: AuthRequest, res) => {
  if (!req.file) { res.status(400).json({ error: 'No audio file' }); return; }

  const userId = req.userId!;
  const expected = req.body.expected as string | undefined;
  const language = (req.body.language as string) || 'uz';

  // Check daily limit
  const today = new Date().toISOString().split('T')[0];
  const admin = getSupabase();

  const { data: usage } = await admin
    .from('transcription_usage')
    .select('count')
    .eq('user_id', userId)
    .eq('date', today)
    .single();

  if (usage && usage.count >= 100) {
    res.status(429).json({ error: 'limit_reached' });
    return;
  }

  // Transcribe
  const audioBuffer = req.file.buffer.buffer.slice(
    req.file.buffer.byteOffset,
    req.file.buffer.byteOffset + req.file.buffer.byteLength
  );
  const result = await transcribeAudio(
    audioBuffer,
    req.file.originalname || 'audio.webm',
    req.file.mimetype || 'audio/webm',
    expected
  );

  if (!result) {
    res.status(503).json({ error: 'transcription_failed' });
    return;
  }

  let { text } = result;
  const normalizedText = text.replace(/\d/g, '').trim();

  // Silence detection
  if (!normalizedText || result.noSpeechProb > 0.6) {
    res.json({ text: '', result: 'no_speech', feedback: '', source: result.source });
    return;
  }

  // Hallucination detection: expected is Chinese but heard is Latin
  if (expected && isChinese(expected) && isLatin(text)) {
    res.json({ text: '', result: 'no_speech', feedback: '', source: result.source });
    return;
  }

  // Post-correction
  if (expected) {
    text = await postCorrect(expected, text);
  }

  // Scoring
  if (!expected) {
    res.json({ text, result: 'correct', feedback: '', source: result.source });
  } else {
    const score = await scoreAnswer(expected, text, language);
    const response: TranscribeResponse = {
      text,
      result: score.result,
      feedback: score.feedback,
      source: result.source,
    };
    res.json(response);
  }

  // Increment usage (fire-and-forget)
  admin.from('transcription_usage').upsert(
    { user_id: userId, date: today, count: (usage?.count || 0) + 1 },
    { onConflict: 'user_id,date' }
  ).then(() => {});
});

export { router as transcribeRouter };
```

```typescript
import { subscriptionRouter } from './routes/subscription.js';
import { healthRouter } from './routes/health.js';

const app = express();

app.use(helmet());
app.use(cors());
app.use(express.json());

// Routes
app.use('/health', healthRouter);
app.use('/auth', authRouter);
app.use('/transcribe', transcribeRouter);
app.use('/', progressRouter);
app.use('/payment', paymentRouter);
app.use('/subscription', subscriptionRouter);

app.listen(config.port, () => {
  console.log(`Blim API running on port ${config.port}`);
});

export { app };
```

## Endpoint Summary

| Method | Path | Auth | Description |
|--------|------|------|-------------|
| GET | /health | No | Server + DB status |
| POST | /auth/google | No | Google Sign-In |
| POST | /auth/telegram | No | Telegram OAuth |
| POST | /auth/refresh | Yes | Refresh JWT |
| POST | /auth/logout | Yes | Delete session |
| POST | /auth/session-check | Yes | Validate nonce |
| POST | /transcribe | Yes | Speech transcription + grading |
| GET | /progress | Optional | Get lesson progress |
| POST | /progress | Yes | Save lesson visit |
| GET | /stars | Optional | Get star ratings |
| POST | /stars | Yes | Save star rating |
| GET | /dialogue-progress | Optional | Get dialogue scores |
| POST | /dialogue-progress | Yes | Save dialogue score |
| POST | /payment/screenshot | Yes | Upload payment screenshot |
| POST | /payment/google | Yes | Verify Google Play purchase |
| GET | /payment/status | Yes | Get payment status |
| GET | /subscription | Optional | Check active subscription |
