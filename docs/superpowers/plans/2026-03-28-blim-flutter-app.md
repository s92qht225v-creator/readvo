**Goal:** Native Android app for Blim Chinese language learning platform, with offline-first content, Google/Telegram auth, and full feature parity with the web app.

**Architecture:** Flutter app with Riverpod state management, Dio HTTP client pointing to the Blim TS API, bundled JSON content, and cached audio. The app never talks directly to Supabase.

```dart
// lib/services/api_client.dart
import 'package:dio/dio.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';
import '../config.dart';

class ApiClient {
  late final Dio dio;
  final _storage = const FlutterSecureStorage();

  ApiClient() {
    dio = Dio(BaseOptions(
      baseUrl: AppConfig.apiBaseUrl,
      connectTimeout: const Duration(seconds: 10),
      receiveTimeout: const Duration(seconds: 30),
    ));

    dio.interceptors.add(InterceptorsWrapper(
      onRequest: (options, handler) async {
        final token = await _storage.read(key: 'access_token');
        if (token != null) {
          options.headers['Authorization'] = 'Bearer $token';
        }
        handler.next(options);
      },
    ));
  }

  Future<void> setToken(String token) async {
    await _storage.write(key: 'access_token', value: token);
  }

  Future<void> clearToken() async {
    await _storage.delete(key: 'access_token');
  }

  Future<String?> getToken() async {
    return _storage.read(key: 'access_token');
  }
}
```

```dart
// lib/services/auth_service.dart
import 'package:google_sign_in/google_sign_in.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';
import 'api_client.dart';
import '../models/user.dart';
import '../models/user.dart';

class AuthService {
  final ApiClient _api;
  final _storage = const FlutterSecureStorage();
  final _googleSignIn = GoogleSignIn();

  AuthService(this._api);

  Future<User?> loginWithGoogle() async {
    final account = await _googleSignIn.signIn();
    if (account == null) return null;

    final auth = await account.authentication;
    if (auth.idToken == null) return null;

    final response = await _api.dio.post('/auth/google', data: {
      'idToken': auth.idToken,
    });

    final token = response.data['access_token'] as String;
    final nonce = response.data['session_nonce'] as String;

    await _api.setToken(token);
    await _storage.write(key: 'session_nonce', value: nonce);

    return User(
      id: '',  // Will be decoded from JWT
      name: account.displayName ?? '',
      email: account.email,
      avatarUrl: account.photoUrl,
    );
  }

  Future<void> logout() async {
    try {
      await _api.dio.post('/auth/logout');
    } catch (_) {}
    await _api.clearToken();
    await _storage.delete(key: 'session_nonce');
    await _googleSignIn.signOut();
  }

  Future<bool> checkSession() async {
    final nonce = await _storage.read(key: 'session_nonce');
    if (nonce == null) return false;

    try {
      final response = await _api.dio.post('/auth/session-check', data: {'nonce': nonce});
```

```dart
// lib/services/content_service.dart
import 'dart:convert';
import 'package:flutter/services.dart';
import '../models/dialogue.dart';
import '../models/flashcard.dart';
import '../models/karaoke.dart';
import '../models/lesson.dart';

class ContentService {
  Future<Map<String, dynamic>> _loadJson(String path) async {
    final string = await rootBundle.loadString(path);
    return json.decode(string) as Map<String, dynamic>;
  }

  Future<List<dynamic>> _loadJsonList(String path) async {
    final string = await rootBundle.loadString(path);
    return json.decode(string) as List<dynamic>;
  }

  // Dialogues
  Future<List<DialogueInfo>> loadDialogueList() async {
    final files = await rootBundle.loadString('AssetManifest.json');
    final manifest = json.decode(files) as Map<String, dynamic>;
    final dialogueFiles = manifest.keys
        .where((k) => k.startsWith('assets/content/dialogues/') && k.endsWith('.json'))
        .toList()
      ..sort();

    final dialogues = <DialogueInfo>[];
    for (final file in dialogueFiles) {
      final data = await _loadJson(file);
      final id = file.split('/').last.replaceAll('.json', '');
      dialogues.add(DialogueInfo.fromJson({...data, 'id': id}));
    }
    return dialogues;
  }

  Future<List<DialogueLine>> loadDialogue(String id) async {
    final data = await _loadJson('assets/content/dialogues/$id.json');
    final sentences = data['sentences'] as List;
    return sentences.map((s) => DialogueLine.fromJson(s)).toList();
  }

  // Flashcards
  Future<List<FlashcardWord>> loadFlashcardDeck(String deckId) async {
    final data = await _loadJsonList('assets/content/flashcards/$deckId.json');
    return data.map((w) => FlashcardWord.fromJson(w)).toList();
  }

  // Karaoke
  Future<KaraokeSong> loadKaraokeSong(String songId) async {
    final data = await _loadJson('assets/content/karaoke/$songId.json');
    return KaraokeSong.fromJson(songId, data);
  }

  // Lessons
  Future<List<Section>> loadLessonPage(int lesson, int page) async {
    final data = await _loadJson('assets/content/lessons/lesson$lesson-page$page.json');
    final sections = data['sections'] as List;
    return sections.map((s) => Section.fromJson(s)).toList();
  }
}
```
